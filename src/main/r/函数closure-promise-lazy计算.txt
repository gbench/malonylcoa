# promise lazy 计算的实例
> f<-function(i) function(x) x^i # 定义函数：i是位于定义f的环境的（），注意这里的i,是共享的,是lazy计算的。
> environment(f) # f 的定义函数
<environment: R_GlobalEnv>
> fs<-list();for(i in 1:5)fs[[i]]<-f(i) # 注意,由于i是lazy计算的promise,for 循环的i变量其是在fs中共享的都是R_GlobalEnv中的i
> i

# promise lazy 计算的实例
> f2<-function(i) {force(i);function(x) x^i} # 定义函数，这里使用force保证每次调用f2都立即计算i,即i不是共享的，禁止lazy计算而是立即计算
> fs2<-list();for(i in 1:5)fs2[[i]]<-f2(i) #生成函数
